// --- Basic game configuration ---
const GAME_DURATION_SECONDS = 30; // Total game time
const MOVE_INTERVAL_MS = 700; // How often the box moves (smaller = harder)

// Cached DOM elements
const scoreEl = document.getElementById("score");
const timeEl = document.getElementById("time");
const startBtn = document.getElementById("startBtn");
const restartBtn = document.getElementById("restartBtn");
const gameArea = document.getElementById("gameArea");
const target = document.getElementById("target");
const gameOverOverlay = document.getElementById("gameOverOverlay");
const finalScoreEl = document.getElementById("finalScore");

// Game state
let score = 0;
let timeRemaining = GAME_DURATION_SECONDS;
let moveIntervalId = null;
let countdownIntervalId = null;
let isRunning = false;

// Simple click sound using Web Audio API (no external files)
let audioContext = null;
function playClickSound() {
  try {
    // Create (or reuse) an audio context the first time we play a sound
    if (!audioContext) {
      audioContext = new (window.AudioContext || window.webkitAudioContext)();
    }

    const duration = 0.08; // seconds
    const oscillator = audioContext.createOscillator();
    const gainNode = audioContext.createGain();

    oscillator.type = "square";
    oscillator.frequency.value = 880; // Hz (pitch)

    // Quick fade out so it sounds more like a "click"
    gainNode.gain.setValueAtTime(0.15, audioContext.currentTime);
    gainNode.gain.exponentialRampToValueAtTime(
      0.001,
      audioContext.currentTime + duration
    );

    oscillator.connect(gainNode);
    gainNode.connect(audioContext.destination);

    oscillator.start();
    oscillator.stop(audioContext.currentTime + duration);
  } catch (e) {
    // If audio fails (e.g. browser restriction), ignore silently
  }
}

// Position the target at a random spot inside the game area
function moveTargetToRandomPosition() {
  const areaRect = gameArea.getBoundingClientRect();
  const targetRect = target.getBoundingClientRect();

  // Compute how far we can move while keeping the whole box visible
  const maxX = areaRect.width - targetRect.width;
  const maxY = areaRect.height - targetRect.height;

  const randomX = Math.random() * maxX;
  const randomY = Math.random() * maxY;

  target.style.transform = `translate(${randomX}px, ${randomY}px)`;
}

// Start or restart the game
function startGame() {
  // Reset state
  score = 0;
  timeRemaining = GAME_DURATION_SECONDS;
  isRunning = true;

  scoreEl.textContent = score;
  timeEl.textContent = timeRemaining.toString();

  // Reset UI
  target.classList.remove("hidden");
  gameOverOverlay.classList.add("hidden");
  startBtn.textContent = "Restart";
  startBtn.disabled = false;

  // Clear existing timers if any (in case of restart)
  if (moveIntervalId !== null) {
    clearInterval(moveIntervalId);
  }
  if (countdownIntervalId !== null) {
    clearInterval(countdownIntervalId);
  }

  // Immediately move target once, then keep moving it on a timer
  moveTargetToRandomPosition();
  moveIntervalId = setInterval(moveTargetToRandomPosition, MOVE_INTERVAL_MS);

  // Countdown timer for the game duration
  countdownIntervalId = setInterval(() => {
    timeRemaining -= 1;
    timeEl.textContent = timeRemaining.toString();

    if (timeRemaining <= 0) {
      endGame();
    }
  }, 1000);
}

// End the game and show the game over screen
function endGame() {
  if (!isRunning) return; // Avoid double-ending

  isRunning = false;

  // Stop moving the target and the timer
  if (moveIntervalId !== null) {
    clearInterval(moveIntervalId);
    moveIntervalId = null;
  }
  if (countdownIntervalId !== null) {
    clearInterval(countdownIntervalId);
    countdownIntervalId = null;
  }

  // Hide the target and show the overlay
  target.classList.add("hidden");
  finalScoreEl.textContent = score.toString();
  gameOverOverlay.classList.remove("hidden");
}

// Handle clicking the moving box
function handleTargetClick(event) {
  if (!isRunning) {
    return;
  }

  // Stop the click from bubbling up to the game area
  event.stopPropagation();

  // Increase score and update the UI
  score += 1;
  scoreEl.textContent = score.toString();

  // Play a short "click" sound
  playClickSound();

  // Move the box immediately on click (feels more responsive)
  moveTargetToRandomPosition();
}

// Optional: if the player taps/clicks empty space while game is running,
// you could deduct points or just ignore. Here we simply ignore.
// gameArea.addEventListener("click", () => { /* no-op */ });

// Wire up buttons and target
startBtn.addEventListener("click", () => {
  // On first click, or any time after a game over, just startGame again
  startGame();
});

restartBtn.addEventListener("click", () => {
  gameOverOverlay.classList.add("hidden");
  startGame();
});

target.addEventListener("click", handleTargetClick);

// Initialize labels with default values
timeEl.textContent = GAME_DURATION_SECONDS.toString();
scoreEl.textContent = "0";
